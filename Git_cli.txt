git init                    #init the git
git clone <url>             #clone project from URL

git status 
git add <option>                              #make the modified selected files ready for staging(commit)
git commit -m "<message>"                     #commit the changes localy
git push                                      #push the commited changes to git repository 


git log                                       #commits' log
git show                                      #last commited commit changes
git diff                                      #show current changes in terminal

git pull                                      #update current branch
git pull <<remote>><<desired_branch>>         #if the current branch is the base of desired_branch or remotely disered_branch this will get the desired_branch commits to the current
git checkout <branch>                         #select branch

git reset --hard <remote>/<desired_branch>            #reset current branch to be in same with remote desired_branch
git merge <desired_branch>                            #merge <branch> to current branch,use it only if the current branch is incoherant with the desired_branch///fix the merge conflicts


git rev-list --count <branch>                 #count all commits of a branch,replace <branch> with --all for all branches
